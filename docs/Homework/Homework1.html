bbWins = function(data, team){
  teamGames = bbTeam(data, team)
  teamGames2 = mutate(teamGames, WIN = (((Home == team) & (Difference > 0)) | ((Away == team) & (Difference < 0))))
  teamGames3 = group_by(teamGames2, WIN)
  teamSummary = summarize(teamGames3, count = n())
  if (! (is.na(teamSummary$count[2]))){
    winLoss = teamSummary$count[2] / (teamSummary$count[1]+teamSummary$count[2]+0)
  }
  else{
    if (teamSummary$WIN[1] == TRUE){
      winLoss = 1
    }
    else{
      winLoss = 0
    }
  }
  return(winLoss)
}
